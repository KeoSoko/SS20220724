export function normalizeMerchantName(name: string): string {
  if (!name) return "";

  return name
    .toLowerCase()
    .replace(/\(.*?\)/g, "")       // remove brackets
    .replace(/via .*$/i, "")       // remove "via Uber Eats"
    .replace(/,.*$/g, "")          // remove address suffix
    .replace(/\s+/g, " ")
    .trim();
}

This ensures merchant rules match variations like:

"Nando's, Malibongwe Drive (via Uber Eats)"
"Nando's Fourways"

nando's

  const result = await db

upsertMerchantCategoryRule

Logic:

if rule exists:
    if category matches → confirmations++
    if category changed → reset confirmations to 1
else:
    create rule with confirmations = 1

Example implementation:

async upsertMerchantCategoryRule(workspaceId: number, merchant: string, category: string) {

  const existing = await db
    .select()
    .from(merchantCategoryRules)
    .where(
      and(
        eq(merchantCategoryRules.workspaceId, workspaceId),
        eq(merchantCategoryRules.merchantPattern, merchant)
      )
    )
    .limit(1);

  if (existing.length > 0) {

    const rule = existing[0];

    if (rule.categoryLabel === category) {
      await db
        .update(merchantCategoryRules)
        .set({
          confirmations: rule.confirmations + 1,
          updatedAt: new Date()
        })
        .where(eq(merchantCategoryRules.id, rule.id));
    } else {
      await db
        .update(merchantCategoryRules)
        .set({
          categoryLabel: category,
          confirmations: 1,
          updatedAt: new Date()
        })
        .where(eq(merchantCategoryRules.id, rule.id));
    }

  } else {

    await db.insert(merchantCategoryRules).values({
      workspaceId,
      merchantPattern: merchant,
      categoryLabel: category,
      confirmations: 1
    });

  }
}
