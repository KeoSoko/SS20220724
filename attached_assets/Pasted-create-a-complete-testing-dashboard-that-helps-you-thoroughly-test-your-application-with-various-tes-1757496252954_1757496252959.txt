    <script>
        // Test status tracking
        const testStatus = {
            'get-test': 'not-tested',
            'post-test': 'not-tested',
            'error-test': 'not-tested',
            'db-write-test': 'not-tested',
            'db-read-test': 'not-tested',
            'perf-test': 'not-tested',
            'load-test': 'not-tested',
            'security-test': 'not-tested',
            'auth-test': 'not-tested',
            'form-test': 'not-tested',
            'ui-test': 'not-tested'
        };
        
        // Update progress bar
        function updateProgress() {
            const totalTests = Object.keys(testStatus).length;
            let completedTests = 0;
            
            for (const test in testStatus) {
                if (testStatus[test] !== 'not-tested') {
                    completedTests++;
                }
            }
            
            const progressPercent = Math.round((completedTests / totalTests) * 100);
            document.getElementById('overall-progress').style.width = progressPercent + '%';
            document.getElementById('progress-text').textContent = progressPercent;
        }
        
        // Run a specific test
        function runTest(testId) {
            const statusElement = document.querySelector(`[onclick="runTest('${testId}')"]`).previousElementSibling.previousElementSibling.querySelector('.status-badge');
            const resultsElement = document.getElementById(`${testId}-results`);
            
            // Update UI to show test is running
            statusElement.className = 'status-badge status-warning';
            statusElement.textContent = 'Testing...';
            resultsElement.style.display = 'block';
            resultsElement.innerHTML = '<div class="result-item">Running test...</div>';
            
            // Simulate test execution (in a real scenario, this would make actual API calls)
            setTimeout(() => {
                // Simulate test results (randomly pass or fail for demonstration)
                const passed = Math.random() > 0.3;
                
                if (passed) {
                    statusElement.className = 'status-badge status-success';
                    statusElement.textContent = 'Passed';
                    resultsElement.innerHTML = `
                        <div class="result-item success">
                            <i class="fas fa-check-circle"></i> Test passed successfully!
                        </div>
                        <div class="result-item">
                            <strong>Details:</strong> All assertions passed. Endpoint is working correctly.
                        </div>
                    `;
                    testStatus[testId] = 'success';
                } else {
                    statusElement.className = 'status-badge status-error';
                    statusElement.textContent = 'Failed';
                    resultsElement.innerHTML = `
                        <div class="result-item error">
                            <i class="fas fa-exclamation-circle"></i> Test failed!
                        </div>
                        <div class="result-item">
                            <strong>Details:</strong> Received status code 500 instead of 200.
                        </div>
                        <div class="result-item">
                            <strong>Recommendation:</strong> Check your server logs for errors.
                        </div>
                    `;
                    testStatus[testId] = 'error';
                }
                
                updateProgress();
            }, 1500);
        }
        
        // Run all tests
        function runAllTests() {
            for (const test in testStatus) {
                runTest(test);
            }
        }
        
        // Reset all tests
        function resetAllTests() {
            for (const test in testStatus) {
                testStatus[test] = 'not-tested';
                const statusElement = document.querySelector(`[onclick="runTest('${test}')"]`).previousElementSibling.previousElementSibling.querySelector('.status-badge');
                statusElement.className = 'status-badge status-not-tested';
                statusElement.textContent = 'Not Tested';
                document.getElementById(`${test}-results`).style.display = 'none';
            }
            
            updateProgress();
        }
        
        // Initialize progress bar
        window.onload = updateProgress;
    </script>
